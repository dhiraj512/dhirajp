## Using Reduce()

```ts title="typescript"
const array = ["apple", "banana", "orange"];

const object = array.reduce((acc, value, index) => {
	acc[`fruit ${index + 1}`] = value;
	return acc;
}, {} as { [key: string]: string });

console.log(object);

// Output: { 'fruit 1': 'apple', 'fruit 2': 'banana', 'fruit 3': 'orange' }
```

## Using fromEntries()

```ts title="typescript"
const array = ["apple", "banana", "orange"];

const object = Object.fromEntries(
	array.map((value, index) => [`fruit${index + 1}`, value])
);

console.log(object);

// Output: { fruit1: "apple", fruit2: "banana", fruit3: "orange" }
```

## Using forEach()

```ts title="typescript"
const array = ["apple", "banana", "orange"];
const object: { [key: string]: string } = {};

array.forEach((value, index) => {
	object[`fruit${index + 1}`] = value;
});

console.log(object);

// Output: { fruit1: "apple", fruit2: "banana", fruit3: "orange" }
```
