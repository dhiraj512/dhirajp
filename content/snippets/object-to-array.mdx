## Array of Keys

```ts title="typescript"
const fruits = {
	fruit1: "apple",
	fruit2: "banana",
	fruit3: "orange",
};

const arrayOfKeys = Object.keys(fruits);
console.log(arrayOfKeys);

// Output: ["fruit1", "fruit2", "fruit3"]
```

## Array of Values

```ts title="typescript"
const fruits = {
	fruit1: "apple",
	fruit2: "banana",
	fruit3: "orange",
};

const arrayOfValues = Object.values(fruits);
console.log(arrayOfValues);

// Output: ["apple", "banana", "orange"]
```

## Array of Keys and Values

```ts title="typescript"
const fruits = {
	fruit1: "apple",
	fruit2: "banana",
	fruit3: "orange",
};

const arrayFromObject = Object.entries(fruits).map(([key, value]) => ({
	key,
	value,
}));

console.log(arrayFromObject);

// Output:
// [
//     { key: 'fruit1', value: 'apple' },
//     { key: 'fruit2', value: 'banana' },
//     { key: 'fruit3', value: 'orange' }
// ]
```
