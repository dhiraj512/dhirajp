### Sorting an array of strings in alphabetical order:

```js
const fruits = ["apple", "banana", "orange", "pear"];
fruits.sort();
console.log(fruits); // Output: ["apple", "banana", "orange", "pear"]
```

### Sorting an array of numbers in ascending order:

```js
const numbers = [5, 2, 9, 1, 7];
numbers.sort((a, b) => a - b);
console.log(numbers); // Output: [1, 2, 5, 7, 9]
```

### Sorting an array of objects by a specific property:

```js
const people = [
	{ name: "John", age: 30 },
	{ name: "Jane", age: 25 },
	{ name: "Bob", age: 40 },
];
people.sort((a, b) => a.age - b.age);
console.log(people); // Output: [{ name: 'Jane', age: 25 }, { name: 'John', age: 30 }, { name: 'Bob', age: 40 }]
```

In this example, we sort the people array of objects by the age property in ascending order.

### Sorting an array of objects using a custom function:

```js
const animals = [
	{ name: "dog", legs: 4 },
	{ name: "spider", legs: 8 },
	{ name: "ant", legs: 6 },
];
animals.sort((a, b) => {
	if (a.legs < b.legs) return -1;
	if (a.legs > b.legs) return 1;
	return 0;
});
console.log(animals); // Output: [{ name: 'dog', legs: 4 }, { name: 'ant', legs: 6 }, { name: 'spider', legs: 8 }]
```

In this example, we sort the animals array of objects by the legs property using a custom sorting function that compares the legs property of each object.
