```js [1, 2, 3]{:js}
const arr = [1, 2, 3, 4, 5];

// Map the array in the forward direction
const forward = arr.map((num) => num * 2);
console.log(forward); // Output: [2, 4, 6, 8, 10]

// Map the array in the opposite direction
const reversed = arr.reverse();
const backward = reversed.map((num) => num * 2);
console.log(backward); // Output: [10, 8, 6, 4, 2]
```
