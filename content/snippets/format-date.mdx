```js
const today = new Date();
const date = [
	today.getFullYear(), //YYYY
	`0${today.getMonth() + 1}`.slice(-2), //MM
	`0${today.getDate()}`.slice(-2), //DD
].join("-");
console.log(date); //YYYY-MM-DD
```
