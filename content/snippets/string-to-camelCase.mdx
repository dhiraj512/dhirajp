```js
function toCamelCase(inputStr) {
	return inputStr
		.trim() // remove whitespace from both sides of a string
		.toLowerCase() // convert string to lowercase letters
		.replace(
			/[-_\s.]+(.)?/g, // match the occurrence of "-", "_", "." or whitespace
			(_, char) => (char ? char.toUpperCase() : "")
		);
}

toCamelCase("hello world");
// or
toCamelCase("hello-world");

// Output: helloWorld
```
